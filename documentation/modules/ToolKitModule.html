import { StartupService, ArlasConfigService, ArlascollaborativesearchService } from './services/startup.services';
import { NgModule, APP_INITIALIZER } from '@angular/core';
import { ConfigService, CollaborativesearchService } from 'arlas-web-core';
import { AppComponent } from './app.component';

export function startupServiceFactory(startupService: StartupService) {
  const load = () => startupService.load();
  return load ;
}
